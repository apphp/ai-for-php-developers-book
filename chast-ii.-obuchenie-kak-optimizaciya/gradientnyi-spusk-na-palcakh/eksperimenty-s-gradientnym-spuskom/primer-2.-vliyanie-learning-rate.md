function train(float $lr): void {

    $x = [1, 2, 3, 4];
    $y = [2, 4, 6, 8];

    $w = 0.0;
    $n = count($x);

    echo PHP_EOL . "Learning rate = $lr" . PHP_EOL;

    for ($epoch = 1; $epoch <= 10; $epoch++) {

        $gradient = 0.0;

        for ($i = 0; $i < $n; $i++) {
            $error = ($w * $x[$i]) - $y[$i];
            $gradient += $x[$i] * $error;
        }

        $gradient = (2 / $n) * $gradient;
        $w -= $lr * $gradient;

        echo "Epoch $epoch: w = " . round($w, 4) . PHP_EOL;
    }
}
